Here is my code for how I used the objects

function animal(species, color, weight, danger, name){
this.species = species;
this.color = color;
this.weight = weight;
this.danger = danger;
this.name = name;

this.statement = function(){
return " You have a " + this.species+ " standing in front of you. It weighs " +
this.weight+ "lbs. and is named " + this.name +". On a danger scale of 1 - 10, it
ranks a " + this.danger+". Don't let the " + this.color+ " color fool you."}
}

var tiger = new animal('Tiger', 'White', 300, 10, 'Snow Flake');
var turtle = new animal('Turtle', 'Green', 5, 1, 'Speed Racer');
var dog = new animal('Dog', 'Black and Brown', 75, 5, 'Lady Bug');

document.getElementById('animalOneDiv').innerHTML=tiger.statement();
document.getElementById('animalTwoDiv').innerHTML=turtle.statement();
document.getElementById('animalThreeDiv').innerHTML=dog.statement();
